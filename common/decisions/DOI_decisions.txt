poland_claims_on_bohemia = {
	is_shown = {
		this = c:PLC
		is_subject = no
		exists = c:AUS
		NOT = { has_global_variable = plc_claimed_bohemia }
		exists = c:AUS	
	}

	possible = {
		this = c:PLC
		c:PLC = {
			has_technology_researched = nationalism
			country_rank >= rank_value:great_power

			NOR = {
				any_scope_state = { state_region = s:STATE_BOHEMIA }
				any_scope_state = { state_region = s:STATE_MORAVIA }
			}
			
		}
		c:AUS = {
			NOT = { has_truce_with = c:PLC }

			# Either AUS is independent OR a subject of PLC (vassal/integration case)
			OR = {
				is_subject = no
				is_subject_of = ROOT
			}

			# Situation in which the play/transfer makes sense
			OR = {
				# Someone is already at war with AUS (opportunity window)
				any_country = { has_war_with = c:AUS }

				# PLC has leverage
				c:PLC = { is_owed_obligation_by = c:AUS }

				# AUS actually holds at least one of the target regions
				any_scope_state = {
					filter = { owner = c:AUS }
					OR = {
						state_region = s:STATE_BOHEMIA
						state_region = s:STATE_MORAVIA
					}
				}
			}
		}
	}

	when_taken = {
		# Mark used and fire the event which branches the outcomes
		set_global_variable = plc_claimed_bohemia
		trigger_event = { id = doi_poland_bohemia.1 popup = yes }
	}

	ai_chance = { base = 15 }
}
