poland_claims_on_hungary = {
	is_shown = {
		this = c:PLC
		is_subject = no
		exists = c:HUN
		NOT = { has_global_variable = plc_claimed_hungary }
	}

	possible = {
		this = c:PLC
		c:PLC = {
			has_technology_researched = romanticism
			has_technology_researched = pan-nationalism
			country_rank >= rank_value:great_power

			# Don't show if PLC already owns a significant chunk (both main cores) — adjust to taste
			NOR = {
				any_scope_state = { state_region=s:STATE_HUNGARY }
				any_scope_state = { state_region=s:STATE_TRANSDANUBIA }
			}
		}

		# Target side sanity: either AUS or HUN is a valid counterpart and no active truce with PLC
		c:HUN = {
			NOT = { has_truce_with = c:PLC }

			OR = {
				is_subject = no
				is_subject_of = ROOT
			}

			# Situation in which the play/transfer makes sense
			OR = {
				# Someone is already at war with HUN (opportunity window)
				any_country = { has_war_with = c:HUN }
				# PLC has leverage
				c:PLC = { is_owed_obligation_by = c:HUN }
			}
		}
				
	}

	when_taken = {
		set_global_variable = plc_claimed_hungary
		trigger_event = { id = doi_poland_hungary.1 popup = yes }
	}

	ai_chance = { base = 12 }
}
