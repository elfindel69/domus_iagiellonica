namespace = doi_france

doi_france.1 = {
	type = country_event # can be country_event, character_event, or state_event
	placement = root
	title = doi_france.1.t
	desc = doi_france.1.d
	flavor = doi_france.1.f
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	immediate = {
		set_variable = kings_death
    }
    
    trigger = {
        exists = c:FRA
        this = c:FRA
		any_scope_character = {
			ruler = {
				NOT = { has_variable = king_louis_philippe }
			}
			
		}
		
		has_law = law_type:law_monarchy
		NOT = {has_variable = kings_death}
	}
	
	duration = 3

	event_image = {
		video = "gfx/event_pictures/unspecific_ruler_speaking_to_people.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/leader_speaking_to_a_group_of_people"

	#towards a new Republic
	option = {
		name = doi_france.1.a
		trigger = {
			or= {
				is_enacting_law = law_type:law_presidential_republic
				is_enacting_law = law_type:law_parliamentary_republic
			}
		}
		if = {
			limit = {
				is_enacting_law = law_type:law_presidential_republic
			}
			activate_law = law_type:law_presidential_republic
		}
		if = {
			limit = {
				is_enacting_law = law_type:law_parliamentary_republic
			}
			activate_law = law_type:law_parliamentary_republic
		}
		# monarchists get VERY angry
		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_monarchy
					value > neutral
				}
			}
			add_modifier = {
				name = regicide
				months = short_modifier_time
			}
		}
    }
	#keeping the monarchy
	option = {
		name = doi_france.1.b
		trigger = {
			nor= {
				is_enacting_law = law_type:law_presidential_republic
				is_enacting_law = law_type:law_parliamentary_republic
			}
		}
		default_option = yes
			# monarchists resist
		every_interest_group = {
			limit = {
				law_stance = {
					law = law_type:law_monarchy
					value > neutral
				}
			}
			add_modifier = {
				name = attracting_notables
				months = normal_modifier_time
			}
		}
	
		
		
    }
}
